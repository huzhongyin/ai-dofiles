   192→    func requestRemoteKeyList(isFromInit: Bool) {
   193→        DigitalKeyManager.requestDigitalKeyList(uid: uid, vehicle: self.vehicle, vin: self.vehicle.vin ?? "", vehiclePreference: self.vehiclePreference) { data in
   194→            self.dealWithRemoteKeys(cards: data?.cardList ?? [], isFromInit: isFromInit, localBleInfoOutsideSDK: nil, noBioVerify: nil, doublesidedAuthCode: nil)
   195→            if let jsonData = try? JSONEncoder().encode(data),
   196→               let jsonDict = try? JSONSerialization.jsonObject(with: jsonData, options: []) as? [String: Any] {
   197→                ProxyManger.shared.dbProxy?.refreshDigitalKeyList?(dataList: jsonDict, uid: self.uid, vin: self.vehicle.vin)
   198→            }
   199→           
   200→            guard let cardList = data?.cardList else {
   201→                self.carKeyPlugin?.remoteCarKeyChanged(keyIdList: nil, uid: self.uid)
   202→                return
   203→            }
   204→            var keyIdList: [String] = [String]()
   205→            for card in cardList {
   206→                if card.cardType == .CCC {
   207→                    keyIdList.append(card.cardNo ?? "")
   208→                }
   209→            }
   210→            if keyIdList.count == 0 {
   211→                self.carKeyPlugin?.remoteCarKeyChanged(keyIdList: nil, uid: self.uid)
   212→            } else {
   213→                self.carKeyPlugin?.remoteCarKeyChanged(keyIdList: keyIdList, uid: self.uid)
   214→            }
   215→        } failed: { error in
   216→            self.failLoadRemoteDigitalKeyList()
   217→            guard self.bleInfo == nil else {
   218→                return
   219→            }
   220→            self.fetchMac()
   221→        }
   222→    }

   239→    func decodeToModel<T: Codable>(dictionary: [String: Any], to type: T.Type) -> T? {
   240→        let jsonData = try? JSONSerialization.data(withJSONObject: dictionary, options: [])
   241→        return jsonData.flatMap { try? JSONDecoder().decode(T.self, from: $0) }
   242→    }

   244→    func dealWithRemoteKeys(cards: [BLEKeyCard]!, isFromInit: Bool!, localBleInfoOutsideSDK: BLEInfo?, noBioVerify:Int?, doublesidedAuthCode: String?) {
   245→        var allKey: [DigitalKey]! = []
   246→        for localKey in self.digitalKeyList ?? []{
   247→            var isFound = false
   248→            for remoteKey in cards {
   249→                if localKey.cardNo == remoteKey.cardNo && localKey.keyCardType == remoteKey.cardType {
   250→                    isFound = true
   251→                    break
   252→                }
   253→                if localKey.keyCardType == .MONA_BLE && remoteKey.cardType == .MONA_BLE, localKey.vin == remoteKey.vin {
   254→                    isFound = true
   255→                    break
   256→                }
   257→            }
   258→            if isFound == false {
   259→                localKey.removeAllKeyMessages()
   260→                localKey.remove()
   261→                var index = 0
   262→                for key: DigitalKey in self.digitalKeyList {
   263→                    let isSupportCarControl = key.isSupportMultiKey()
   264→                    if ((!isSupportCarControl && key.keyCardType == localKey.keyCardType) ||
   265→                        isSupportCarControl && key.cardNo == localKey.cardNo)
   266→                        {
   267→                        self.digitalKeyList.remove(at: index)
   268→                        break;
   269→                    }
   270→                    index = index + 1
   271→                }
   272→                self.repareBleInfo()
   273→                self.delegate?.didRemoveKey?(vin: self.vehicle.vin, keyCardType: localKey.keyCardType, success: true)
   274→            }
   275→        }
   276→        
   277→        for remoteKey in cards {
   278→            var isFound = false
   279→            var targetKey: DigitalKey?
   280→            for localKey in self.digitalKeyList {
   281→
   282→                if (localKey.keyCardType == .MONA_BLE && remoteKey.cardType == .MONA_BLE) ||
   283→                            
   284→                (localKey.cardNo == remoteKey.cardNo && localKey.keyCardType == remoteKey.cardType)
   285→                {
   286→                    isFound = true
   287→                    localKey.bleKeyCard = remoteKey
   288→                    localKey.save()
   289→                    allKey .append(localKey)
   290→                    targetKey = localKey
   291→                    break
   292→                }
   293→            }
   294→            if isFound == false {
   295→                let digitalKey = DigitalKey(bleKeyCard: remoteKey, uid:self.uid, vin: remoteKey.vin ?? "", keyCardType: remoteKey.cardType ?? KeyCardType.NONE, cardNo: remoteKey.cardNo)
   296→                digitalKey.doublesidedAuthCode = doublesidedAuthCode
   297→                digitalKey.bleKeyCard = remoteKey
   298→                if digitalKey.keyCardType == KeyCardType.SAFE {
   299→                    if ProxyManger.shared.dbProxy?.checkKeyExist?(cardNo: digitalKey.cardNo) == true {
   300→                        digitalKey.isRegisterInThisInstall = true
   301→                    }
   302→                    if noBioVerify == 1 {
   303→                        digitalKey.noBioVerfy = true
   304→                    } else {
   305→                        digitalKey.noBioVerfy = false
   306→                    }
   307→                }
   308→                digitalKey.save()
   309→                targetKey = self.appendDigitalKey(digitalKey: digitalKey)
   310→            }
   311→            if targetKey?.supportKeyMessage() == true {
   312→                targetKey?.downloadKeyMessages(callBack: { isSuc in
   313→                    if targetKey?.supportDoublesidedAuth() == true && targetKey?.doublesidedAuthCode == nil {
   314→                        self.downloadDoubleSideCode { isSuc, code in
   315→                            if isSuc == true {
   316→                                targetKey?.doublesidedAuthCode = code
   317→                                targetKey?.save()
   318→                            }
   319→                            self.downloadMacAddress(cards: cards, targetKey: targetKey, isFromInit: isFromInit, localBleInfoOutsideSDK: localBleInfoOutsideSDK)
   320→                        }
   321→                    } else {
   322→                        self.downloadMacAddress(cards: cards, targetKey: targetKey, isFromInit: isFromInit, localBleInfoOutsideSDK: localBleInfoOutsideSDK)
   323→                    }
   324→                })
   325→            } else {
   326→                self.downloadMacAddress(cards: cards, targetKey: targetKey, isFromInit: isFromInit, localBleInfoOutsideSDK: localBleInfoOutsideSDK)
   327→            }
   328→        }
   329→    }

   331→    func downloadMacAddress(cards: [BLEKeyCard]!, targetKey: DigitalKey?, isFromInit: Bool!, localBleInfoOutsideSDK: BLEInfo?) {
   332→        if targetKey?.isNeedMacAddress() == false {
   333→            self.finishLoadKeys(cards: cards)
   334→            return;
   335→        }
   336→        if isFromInit == false {
   337→            if let bleInfo = targetKey?.bleInfo {
   338→                if bleInfo.macAddress != nil {
   339→                    self.finishLoadKeys(cards: cards)
   340→                    return;
   341→                }
   342→            }
   343→        }
   344→        let responseCallback: BLEKeyHttpRequestCallback<BLEInfo> = { response in
   345→            if (response.code == 200 && response.data != nil) {
   346→                let info = response.data!
   347→                self.receiveMacAddress(info: info, isLocal: false, targetKey: targetKey)
   348→            } else {
   349→                if localBleInfoOutsideSDK != nil {
   350→                    self.receiveMacAddress(info: localBleInfoOutsideSDK ?? BLEInfo(), isLocal: true, targetKey: targetKey)
   351→                } else {
   352→                    if let digitalKey: DigitalKey = self.validBlePrivateKey() {
   353→                        if digitalKey.bleInfo?.macAddress != nil {
   354→                            self.bleInfo = digitalKey.bleInfo
   355→                        } else {
   356→                            self.fetchMac()
   357→                        }
   358→                    } else {
   359→                        self.fetchMac()
   360→                    }
   361→                }
   362→            }
   363→            self.finishLoadKeys(cards: cards)
   364→        }
   365→        BLEKeyHttpRequest.postRequest(path: BLEKeyHttpRequestPath.bleInfo, payload: ["vin": targetKey?.vin ?? ""], callback: responseCallback)
   366→    }

   368→    func receiveMacAddress(info :BLEInfo, isLocal: Bool, targetKey: DigitalKey?) {
   369→        let macAddress = (info.macAddress ?? "") as String
   370→        let preText = isLocal == true ? "app旧版" : "下载到"
   371→        Log.i(tag:"DigitalKeyManager", "\(preText)Mac地址 macAddress:", macAddress)
   372→        if (self.vehicle.vehicleTypeCode != "ED") {
   373→            info.useScanUUIDs = true
   374→        }
   375→        targetKey?.bleInfo = info
   376→        self.bleInfo = info
   377→        targetKey?.save()
   378→        let klist = DigitalKey.queryDigitalKeyList(vin: targetKey?.vin ?? "")
   379→        Log.i(tag:"DigitalKeyManager", "数字钥匙写入后加载结果: ", klist)
   380→    }

   382→    func downloadDoubleSideCode(callBack: @escaping (_ isSuc: Bool, _ code: String?) -> Void) {
   383→        let responseCallback: BLEKeyHttpRequestCallback<requestVehicleVerifyCodeResponse> = { verifyResponse in
   384→            guard verifyResponse.code == 200 else {
   385→                callBack(false, nil)
   386→                return
   387→            }
   388→            if let data: requestVehicleVerifyCodeResponse = verifyResponse.data {
   389→                callBack(true, data.verifyCodes.phoneCode.code)
   390→            } else {
   391→                callBack(false, nil)
   392→            }
   393→        }
   394→        BLEKeyHttpRequest.postRequest(path: BLEKeyHttpRequestPath.requestVehicleVerifyCode, payload:["typeList":[1, 2],"vin":self.vehicle.vin ?? ""],callback: responseCallback)
   395→    }

   494→    func vaildBleKeyCategory() -> DigitalKeyCategory {
   495→        var fallbackCategory: DigitalKeyCategory {
   496→            return validBlePrivateKey()?.keyCategory ?? .NONE
   497→        }
   498→        
   499→        guard let communication = self.communication,
   500→              let cccVid = self.vehiclePreference.extInfo?.cccVid,
   501→              let carKeyManager = self.carKeyPlugin else {
   502→            return fallbackCategory
   503→        }

   623→    public func createKey(keyCardType: KeyCardType, success: @escaping VehicleDigitalKeySuccessResponseCallback<DigitalKey>, failed: @escaping VehicleFailedCallback<VehicleError>) {
   624→        let key: DigitalKey? = self.digitalKey(keyCardType: keyCardType)
   625→        if (key != nil && key?.isValidKey() == true) {
   626→            Log.i(tag:"DigitalKeyManager", "数字钥匙创建 \(keyCardType) createKey 该类型的钥匙已经存在，不允许重复创建")
   627→            let resp = VehicleError(code: .EnvironmentError, desc: "执行失败(重复创建)")
   628→            failed(resp)
   629→            return
   630→        }
   631→        
   632→        let plugin = self.createPlugin(keyCardType: keyCardType)
   633→        if (plugin == nil) {
   634→            let resp = VehicleError(code: .NotSupported, desc: "尚未支持，操作失败")
   635→            failed(resp)
   636→            return
   637→        }
   638→
   639→        self.delegate?.willCreateKey?(vin: self.vehicle.vin, keyCardType: keyCardType)
   640→        self.createKeyPlugin = plugin
   641→
   642→        plugin!.createKey(success: { data in
   643→            if (data != nil && data!.digitalKeyList != nil && (data!.digitalKeyList.count ?? 0) > 0) {
   644→                for digitalKeyItem in data!.digitalKeyList{
   645→                    digitalKeyItem.save()
   646→                    let digitalKeyAppend = self.appendDigitalKey(digitalKey: digitalKeyItem)
   647→                    if (digitalKeyAppend != nil && data!.authDataList != nil) {
   648→                        digitalKeyAppend!.saveAuthDataList(dataList: data!.authDataList!)
   649→                    }
   650→                }
   651→                var digitalKey: DigitalKey?
   652→                if data!.digitalKeyList.count > 1 {
   653→                    digitalKey = data!.digitalKeyList[1]
   654→                } else {
   655→                    digitalKey = data!.digitalKeyList[0]
   656→                }
   657→                self.delegate?.didCreateKey?(vin: self.vehicle.vin, keyCardType: keyCardType, success: true)
   658→                Log.i(tag:"DigitalKeyManager", "创建钥匙后当前报文数量:", digitalKey?.keyMessageCount())
   659→
   660→                let test: BLEKeyHttpRequestCallback<BLEInfo> = { response in
   661→                    if (response.code == 200 && response.data != nil) {
   662→                        let info = response.data!
   663→                        let macAddress = (info.macAddress ?? "") as String
   664→                        Log.i(tag:"DigitalKeyManager", "下载到Mac地址 macAddress:", macAddress)
   665→                        if (self.vehicle.vehicleTypeCode != "ED") {
   666→                            info.useScanUUIDs = true
   667→                        }
   668→                        digitalKey?.bleInfo = info
   669→                        self.bleInfo = info // 业务方会监听 self.bleInfo 变化，重连蓝牙
   670→                        let klist = DigitalKey.queryDigitalKeyList(vin: self.vehicle.vin!)
   671→                        Log.i(tag:"DigitalKeyManager", "数字钥匙写入后加载结果: ", klist)
   672→                        if digitalKey?.bleKeyCard != nil && digitalKey?.bleKeyCard?.status == XPBLEKeyCardStatus.XPBLEKeyCardStatusReadyToActivate {
   673→                            digitalKey?.bleKeyCard?.status = XPBLEKeyCardStatus.XPBLEKeyCardStatusToBeActivated
   674→                            if let saveKey = digitalKey {
   675→                                saveKey.save()
   676→                                let digitalKeyAppend = self.appendDigitalKey(digitalKey: saveKey)
   677→                            }
   678→                            self.createKey(keyCardType: KeyCardType.MONA_BLE, success: success, failed: failed)
   679→                        }else {
   680→                            if let saveKey = digitalKey {
   681→                                saveKey.save()
   682→                                let digitalKeyAppend = self.appendDigitalKey(digitalKey: saveKey)
   683→                            }
   684→                            success(digitalKey)
   685→                        }
   686→                    } else {
   687→                        self.fetchMac()
   688→                        success(digitalKey)
   689→                    }
   690→                }
   691→                BLEKeyHttpRequest.postRequest(path: BLEKeyHttpRequestPath.bleInfo, payload: ["vin": self.vehicle.vin ?? ""], callback: test)
   692→            } else {
   693→                self.delegate?.didCreateKey?(vin: self.vehicle.vin, keyCardType: keyCardType, success: false)
   694→                let resp = VehicleError(code: .UnknownError, desc: "执行失败")
   695→                failed(resp)
   696→            }
   697→            self.createKeyPlugin = nil
   698→        }, failed: { error in
   699→            if (error.digitalKey != nil) {
   700→                error.digitalKey!.save()
   701→                _ = self.appendDigitalKey(digitalKey: error.digitalKey!)
   702→            }
   703→            self.delegate?.didCreateKey?(vin: self.vehicle.vin, keyCardType: keyCardType, success: false)
   704→            failed(VehicleError(code: VehicleErrorCode.UnknownError, msg: error.msg, desc: error.desc))
   705→            self.createKeyPlugin = nil
   706→        })
   707→    }

   720→    func startTimerMacAddress(currentVin: String) {
   721→        let timer = Timer(timeInterval: 5, repeats: true) { [weak self] _ in
   722→            if let digitalKey: DigitalKey = self?.validBlePrivateKey() {

   736→    func requestMacAddressOnce(currentVin: String, callBack: ((_ isSuc: Bool) -> Void)?) {
   737→        let responseCallback: BLEKeyHttpRequestCallback<BLEInfo> = { response in
   738→            if (response.code == 200 && response.data != nil) {
   739→                callBack?(true)
   740→                let info = response.data!
   741→                guard let macAddress: String = info.macAddress as? String else {
   742→                    return
   743→                }
   744→                self.fetchMacTimer?.invalidate()
   745→                self.fetchMacTimer = nil
   746→                Log.i(tag:"DigitalKeyManager", "定时器下载到Mac地址:%@ vin:%@", macAddress, currentVin)
   747→                var currentKey: DigitalKey? = self.validBlePrivateKey()
   748→                if currentKey == nil {
   749→                    let digitalKeyList = DigitalKey.queryDigitalKeyList(vin: currentVin)
   750→                    for digitalKey: DigitalKey in digitalKeyList {
   751→                        if digitalKey.vin == currentVin {
   752→                            currentKey = digitalKey
   753→                        }
   754→                    }
   755→                }
   756→                guard let targetKey: DigitalKey = currentKey else {
   757→                    return
   758→                }
   759→                
   760→                if targetKey.bleInfo?.macAddress != nil {
   761→                    self.bleInfo = info
   762→                    return
   763→                }
   764→                targetKey.bleInfo = info
   765→                targetKey.save()
   766→                self.appendDigitalKey(digitalKey: targetKey)
   767→                self.bleInfo = info
   768→            } else {
   769→                callBack?(false)
   770→            }
   771→        }
   772→        BLEKeyHttpRequest.postRequest(path: BLEKeyHttpRequestPath.bleInfo, payload: ["vin": currentVin], callback: responseCallback)
   773→    }

   891→    @objc func refreshDigitalKeyFromNotification(noti: Notification) {
   892→        let code: Int = noti.userInfo?["code"] as? Int ?? 0
   893→        switch code {
   894→        case 0xF2:  // 钥匙已拉黑
   895→            NotificationCenter.default.post(name: VehicleNotification.DigitalKeyCreateKey, object: nil, userInfo: ["type": CreateKeyChangeType.ban.rawValue])
   896→            break
   897→        case 0xF3:  // 钥匙激活超过一天有效期
   898→            NotificationCenter.default.post(name: VehicleNotification.DigitalKeyCreateKey, object: nil, userInfo: ["type": CreateKeyChangeType.oneDay.rawValue])
   899→            break
   900→        case 0xF4:  // vin校验不对
   901→            NotificationCenter.default.post(name: VehicleNotification.DigitalKeyCreateKey, object: nil, userInfo: ["type": CreateKeyChangeType.vin.rawValue])
   902→            break
   903→        case 0xF5:  // 钥匙过期（使用期限）
   904→            NotificationCenter.default.post(name: VehicleNotification.DigitalKeyCreateKey, object: nil, userInfo: ["type": CreateKeyChangeType.disabled.rawValue])
   905→            break
   906→        case 0xF6:  // 钥匙已满
   907→            NotificationCenter.default.post(name: VehicleNotification.DigitalKeyCreateKey, object: nil, userInfo: ["type": CreateKeyChangeType.full.rawValue])
   908→            break
   909→        case 0xF7:  // 钥匙信息签名校验不通过
   910→            NotificationCenter.default.post(name: VehicleNotification.DigitalKeyCreateKey, object: nil, userInfo: ["type": CreateKeyChangeType.otherError.rawValue])
   911→            break
   912→        case 0xF8:  // 钥匙信息不匹配
   913→            NotificationCenter.default.post(name: VehicleNotification.DigitalKeyCreateKey, object: nil, userInfo: ["type": CreateKeyChangeType.otherError.rawValue])
   914→            break
   915→        default:
   916→            break
   917→        }
   918→        self.requestRemoteKeyList(isFromInit: false)
   919→    }
